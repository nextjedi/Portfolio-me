Phase 4: Medium Integration Service (Day 8)

// medium.service.ts import { Injectable } from '@angular/core'; import { HttpClient } from '@angular/common/http'; import { Observable } from 'rxjs'; import { map } from 'rxjs/operators'; export interface BlogPost { title: string; link: string; description: string; pubDate: string; thumbnail?: string; categories?: string[]; author: string; } @Injectable({ providedIn: 'root' }) export class MediumService { private readonly RSS_TO_JSON_API = 'https://api.rss2json.com/v1/api.json'; private readonly MEDIUM_RSS_URL = 'https://medium.com/@arunabhpriyadarshi/feed'; constructor(private http: HttpClient) { } getBlogPosts(): Observable<BlogPost[]> { const url = `${this.RSS_TO_JSON_API}?rss_url=${encodeURIComponent(this.MEDIUM_RSS_URL)}`; return this.http.get<any>(url).pipe( map(response => { if (response.status === 'ok') { return response.items.slice(0, 6).map((item: any) => ({ title: item.title, link: item.link, description: this.cleanDescription(item.description || item.content), pubDate: new Date(item.pubDate).toLocaleDateString(), thumbnail: item.thumbnail || this.extractImageFromContent(item.content), categories: item.categories || [], author: item.author || 'Arunabh Priyadarshi' })); } return []; }) ); } private cleanDescription(description: string): string { // Remove HTML tags and limit length const div = document.createElement('div'); div.innerHTML = description; const text = div.textContent || div.innerText || ''; return text.length > 150 ? text.substring(0, 150) + '...' : text; } private extractImageFromContent(content: string): string { // Extract first image from Medium content const imgRegex = /<img[^>]+src="([^">]+)"/; const match = content.match(imgRegex); return match ? match[1] : 'assets/images/blog-placeholder.jpg'; } }
